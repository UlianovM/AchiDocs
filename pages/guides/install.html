loadkeys ru
#
setfont cyr-sun16

Время и часовой пояс

#
timedatectl set-timezone Europe/Moscow
#
timedatectl status

Разметка дисков

#
free -gh
#
lsblk
#

#

Форматирование разделов

#
mkfs.fat -F32 /dev/boot-раздел -n label
#
mkfs.ext4 /dev/root-раздел -L label
#
mkswap /dev/swap-раздел -L label

Монтирование разделов

#
mount /dev/root-раздел /mnt
#
mount /dev/boot-раздел /mnt/boot --mkdir
#
swapon /dev/swap-раздел

Настройка зеркал и установка базовой системы

#
reflector -c Russia \ -n 8 \ -p https \ --sort rate \ --download-timeout 1 \ --connection-timeout 1 \ --save /etc/pacman.d/mirrorlist
#
pacstrap -P /mnt linux base packages

Генерация fstab

#
genfstab -U /mnt > /mnt/etc/fstab

Вход в новую систему (chroot)

#
arch-chroot /mnt /bin/shell
